## Usage

Add the Toaster component
```tsx
import { Toaster } from "@/components/ui/sonner"

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <head />
      <body>
        <Toaster />
        <main>{children}</main>
      </body>
    </html>
  )
}
```

```tsx
import { toast } from "sonner"
```

```tsx
toast("Event has been created.")
```
